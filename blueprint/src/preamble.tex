% Blueprint preamble — loaded by both print.tex and web.tex
%
% This file configures the leanblueprint plasTeX macros:
%   \lean{decl}    — link to a Lean declaration
%   \leanok        — mark proof as sorry-free
%   \uses{labels}  — declare mathematical dependencies
%   \proves{label} — link a proof block to its statement

\usepackage{amsmath,amssymb,amsthm}
\usepackage{hyperref}
\usepackage{leanblueprint}

% ── Theorem environments ────────────────────────────────────────────
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{corollary}[theorem]{Corollary}
\newtheorem{conjecture}[theorem]{Conjecture}
% `algorithmblock` (theorem-like) — separate from the `algorithm`
% float package used for pseudocode floats elsewhere.
\newtheorem{algorithmblock}[theorem]{Algorithm}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}
\newtheorem{example}[theorem]{Example}
\newtheorem{remark}[theorem]{Remark}
